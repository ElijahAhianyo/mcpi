def testVec3():
    it = Vec3(1, -2, 3)
    assert it.x == 1
    assert it.y == -2
    assert it.z == 3
    assert it.x != -1
    assert it.y != +2
    assert it.z != -3
    clone = it.clone()
    assert it == clone
    it.x += 1
    assert it != clone
    a = Vec3(10, -3, 4)
    b = Vec3(-7, 1, 2)
    c = a + b
    assert c - a == b
    assert c - b == a
    assert a + a == a * 2
    assert a - a == Vec3(0, 0, 0)
    assert a + -a == Vec3(0, 0, 0)
    e = eval(repr(it))
    assert e == it
